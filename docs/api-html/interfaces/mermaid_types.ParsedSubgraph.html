    Interface ParsedSubgraph

    Parsed subgraph (group/fence)

    Represents a logical grouping of nodes in the flowchart. Subgraphs can have their own direction and are used for visual organization.

    Example: Subgraph with Direction

    const subgraph: ParsedSubgraph = {
      id: 'frontend',
      title: 'Frontend Services',
      direction: 'LR',
      nodeIds: ['ui', 'state', 'router']
    };

    Example: Mermaid Subgraph

    %%{init:{"theme":"dark"}}%% flowchart TB subgraph "Backend Services" direction LR api[API] db[(Database)] cache[(Cache)] end
    %%{init:{"theme":"default"}}%% flowchart TB subgraph "Backend Services" direction LR api[API] db[(Database)] cache[(Cache)] end
    flowchart TB
  subgraph "Backend Services"
    direction LR
    api[API]
    db[(Database)]
    cache[(Cache)]
  end
